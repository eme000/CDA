) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

  CONSTRAINT `GESTION_STOCK_ibfk_1` FOREIGN KEY (`ID_AM`) REFERENCES `PRODUIT` (`ID_AM`)

DROP TABLE IF EXISTS `PRODUIT`;
